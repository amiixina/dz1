$ git --version(перевірити чи встановлен git)
$ git clone link (скопіювати проєкт)
cd назва папки (зайти в папку)
ls -a (показуе всі файли)
$ git --help (всі доступні команди та їх значення)
$ git config --global user.name "..." (встановлення імені)
$ git config --global user.email "..." (додати email)
$ git add "name of file" (назва файлу без лапок.додати файл)
$ git status (подивитись зміни)
$ git commit -m "…" (додати коментар. замість … треба написати яку інформацію ми додали)
$ git add .(додати всю інформацію яку ми зробили)
$ git log (внесені зміни)
$ git push origin назва гілки (оновити на хмарному сервісі)
$ git branch (подивитись гілку)
$ git checkout -b назва (зробити нову гілку)
$ git checkout (переключити гілку)
$ git pull origin branch (залити зміни з хмарного сховища)
$ git fetch (залити всі зміни з хмарного сховища)
$ git init(згенерувати папку git та gitignore)
